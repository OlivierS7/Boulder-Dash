30  	Image LittleDiamond;
31  
32  	Image Clock;

51  	}

54  		this.repaint();
55  	}

57  	public Image readImageContour() {
58  		try {
59  			this.Contour = ImageIO.read(new File("sprite/Contour.png"));
60  		} catch (IOException e) {
61  			e.printStackTrace();
62  		}
63  		return this.Contour;
64  	}

71  		}
72  		return this.LittleDiamond;
73  	}

76  		try {
77  			this.Clock = ImageIO.read(new File("sprite/Clock.png"));
78  		} catch (IOException e) {
79  			e.printStackTrace();
80  		}
81  		return this.Clock;
82  	}

84  	@Override
85  	protected void paintComponent(final Graphics graphics) {
86  		Player playerPos = this.viewFrame.getModel().getMap().getPlayer();
87  		graphics.clearRect(0, 0, this.getWidth(), this.getHeight());
88  		if (this.viewFrame.getModel().getMap().getPlayer() != null) {
89  			((Graphics2D) graphics).translate(-playerPos.getX() * scale * pixelImage + this.getWidth() / 2,
90  					(-playerPos.getY()) * scale * pixelImage + this.getHeight() / 2 - 160);
91  			((Graphics2D) graphics).scale(scale, scale);
92  		}
93  		this.readImageContour();
94  		this.readImageLittleDiamond();
95  		this.readImageClock();
96  		Entity[][] entity = viewFrame.getModel().getMap().getEntityMap();
97  		for (int x = 0; x < this.viewFrame.getModel().getMap().getWidthMap(); x++) {
98  			for (int y = 0; y < this.viewFrame.getModel().getMap().getHeightMap(); y++) {
99  				if (entity[x][y] != null) {
100 					graphics.drawImage(entity[x][y].getSprite().getImage(), x * pixelImage, 45 + y * pixelImage, this);
101 				}
102 			}
103 		}
104 		drawGraphics(graphics);
105 		verifyWinOrLose(graphics);
106 	}
107 
108 	private void verifyWinOrLose(Graphics graphics) {
109 		Player playerPos = this.viewFrame.getModel().getMap().getPlayer();
110 		if (this.viewFrame.getModel().getIsTime()) {
111 			this.viewFrame.buildViewFrame(this.getViewFrame().getModel(), 1);
112 			this.viewFrame.revalidate();
113 			if (isWin) {
114 				this.isWin = false;
115 				this.viewFrame.printMessage(this.viewFrame.getModel().getLostTimeMessage());
116 			}
117 		}
118 		if (this.viewFrame.getModel().getIsWin()) {
119 			this.viewFrame.buildViewFrame(this.getViewFrame().getModel(), 2);
120 			this.viewFrame.revalidate();
121 			if (isWin) {
122 				this.isWin = false;
123 				this.viewFrame.printMessage(this.viewFrame.getModel().getWinMessage());
124 			}
125 		}
126 		if (playerPos != null) {
127 			if (!this.viewFrame.getModel().getIsAlive()) {
128 				this.viewFrame.buildViewFrame(this.getViewFrame().getModel(), 1);
129 				this.viewFrame.revalidate();
130 				if (isWin) {
131 					this.isWin = false;
132 					this.viewFrame.printMessage(this.viewFrame.getModel().getLostDeadMessage());
133 				}
134 			}
135 		}
136 	}
137 
138 	public void drawGraphics(Graphics graphics) {
139 		Player playerPos = this.viewFrame.getModel().getMap().getPlayer();
140 		if (playerPos != null) {
141 			((Graphics2D) graphics).scale(1 / scale, 1 / scale);
142 			((Graphics2D) graphics).translate(+playerPos.getX() * scale * pixelImage - this.getWidth() / 2,
143 					(+playerPos.getY()) * scale * pixelImage - this.getHeight() / 2 + 160);
144 			graphics.drawImage(this.Contour, 0, 0, 200, 80, this);
145 			graphics.drawImage(this.Contour, 1395, 0, 200, 80, this);
146 			graphics.drawImage(this.LittleDiamond, 35, 20, 40, 40, this);
147 			graphics.drawImage(this.Clock, 1425, 20, 40, 40, this);
148 			graphics.setFont(new Font("SansSerif", Font.PLAIN, 42));
149 			graphics.drawString(": " + String.valueOf(this.viewFrame.getModel().getMap().getCount()), 85, 55);
150 			graphics.drawString(": " + String.valueOf(this.viewFrame.getModel().getMap().getCountdown()
151 					- this.viewFrame.getModel().getTimeSinceStart() / 1000), 1475, 55);
152 		}
153 	}
